```bash
http {
    server {
         # 可以配置多项也可以组合配置，下面这个是组合配置，一般都是先拒绝哪些IP，再开放哪些网段
         # 注意：下面的配置可以配置在server或http段
        location / {
            # 拒绝特定IP访问
            deny 192.168.1.1;
            # 开放特定IP的网段访问     
            allow 192.168.1.0/24;
            # 开放特定IP的网段访问（注意：下面这个是IPV6的写法）  
            allow 2001:0db8::/32;
            # 拒绝所有访问
            #deny all;
        }
    }
}
```
